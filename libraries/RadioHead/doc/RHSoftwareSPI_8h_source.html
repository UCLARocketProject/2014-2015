13 /// \brief Encapsulate a software SPI interface
14 ///
15 /// This concrete subclass of RHGenericSPI enapsulates a bit-banged software SPI interface.
16 /// Caution: this software SPI interface will be much slower than hardware SPI on most
17 /// platforms.
18 ///
19 /// \par Usage
20 ///
21 /// Usage varies slightly depending on what driver you are using.
22 ///
23 /// For RF22, for example:
24 /// \code
25 /// #include <RHSoftwareSPI.h>
26 /// RHSoftwareSPI spi;
27 /// RH_RF22 driver(SS, 2, spi);
28 /// RHReliableDatagram(driver, CLIENT_ADDRESS);
29 /// void setup()
30 /// {
31 /// spi.setPins(6, 5, 7); // Or whatever SPI pins you need
32 /// ....
33 /// }
34 /// \endcode
35 class RHSoftwareSPI : public RHGenericSPI

39  /// Constructor
40  /// Creates an instance of a bit-banged software SPI interface.
41  /// Sets the SPI pins to the defaults of
42  /// MISO = 12, MOSI = 11, SCK = 13. If you need other assigments, call setPins() before
43  /// calling manager.init() or driver.init().
44  /// \param[in] frequency One of RHGenericSPI::Frequency to select the SPI bus frequency. The frequency
45  /// is mapped to the closest available bus frequency on the platform. CAUTION: the achieved
46  /// frequency will almost certainly be very much slower on most platforms. eg on Arduino Uno, the
47  /// the clock rate is likely to be at best around 46kHz.
48  /// \param[in] bitOrder Select the SPI bus bit order, one of RHGenericSPI::BitOrderMSBFirst or
49  /// RHGenericSPI::BitOrderLSBFirst.

51  RHSoftwareSPI(Frequency frequency = Frequency1MHz, BitOrder bitOrder = BitOrderMSBFirst, DataMode dataMode = DataMode0);
