18 /// \class RHGenericDriver RHGenericDriver.h <RHGenericDriver.h>
19 /// \brief Abstract base class for a RadioHead driver.
20 ///
21 /// This class defines the functions that must be provided by any RadioHead driver.
22 /// Different types of driver will implement all the abstract functions, and will perhaps override
23 /// other functions in this subclass, or perhaps add new functions specifically required by that driver.
24 /// Do not directly instantiate this class: it is only to be subclassed by driver classes.
25 ///
26 /// Subclasses are expected to implement a half-duplex, unreliable, error checked, unaddressed packet transport.
27 /// They are expected to carry a message payload with an appropriate maximum length for the transport hardware
28 /// and to also carry unaltered 4 message headers: TO, FROM, ID, FLAGS
29 ///
30 /// \par Headers
31 ///
32 /// Each message sent and received by a RadioHead driver includes 4 headers:
33 /// -TO The node address that the message is being sent to (broadcast RH_BROADCAST_ADDRESS (255) is permitted)
34 /// -FROM The node address of the sending node
35 /// -ID A message ID, distinct (over short time scales) for each message sent by a particilar node
36 /// -FLAGS A bitmask of flags. The most significant 4 bits are reserved for use by RadioHead. The least
37 /// significant 4 bits are reserved for applications.

49  RHModeIdle, ///< Transport is idle.
50  RHModeTx, ///< Transport is in the process of transmitting a message.
51  RHModeRx ///< Transport is in the process of receiving a message.
52  } RHMode;

62  /// Tests whether a new message is available
63  /// from the Driver.
64  /// On most drivers, if there is an uncollected received message, and there is no message
65  /// currently bing transmitted, this will also put the Driver into RHModeRx mode until
66  /// a message is actually received by the transport, when it will be returned to RHModeIdle.
67  /// This can be called multiple times in a timeout loop.

71  /// Turns the receiver on if it not already on.
72  /// If there is a valid message available, copy it to buf and return true
73  /// else return false.
74  /// If a message is copied, *len is set to the length (Caution, 0 length messages are permitted).
75  /// You should be sure to call this function frequently enough to not miss any messages
76  /// It is recommended that you call it in your main loop.
77  /// \param[in] buf Location to copy the received message
78  /// \param[in,out] len Pointer to available space in buf. Set to the actual number of octets copied.
79  /// \return true if a valid message was copied to buf
80  virtual bool recv(uint8_t* buf, uint8_t* len) = 0;

82  /// Waits until any previous transmit packet is finished being transmitted with waitPacketSent().
83  /// Then loads a message into the transmitter and starts the transmitter. Note that a message length
84  /// of 0 is NOT permitted. If the message is too long for the underlying radio technology, send() will
85  /// return false and will not send the message.
86  /// \param[in] data Array of data to be sent
87  /// \param[in] len Number of bytes of data to send (> 0)

115  /// Sets the address of this node. Defaults to 0xFF. Subclasses or the user may want to change this.
116  /// This will be used to test the adddress in incoming messages. In non-promiscuous mode,
117  /// only messages with a TO header the same as thisAddress or the broadcast addess (0xFF) will be accepted.
118  /// In promiscuous mode, all messages will be accepted regardless of the TO header.
119  /// In a conventional multinode system, all nodes will have a unique address
120  /// (which you could store in EEPROM).
121  /// You would normally set the header FROM address to be the same as thisAddress (though you dont have to,
122  /// allowing the possibilty of address spoofing).
